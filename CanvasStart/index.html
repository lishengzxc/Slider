创建Canvas


							

window.onload = function () {
	var canvas = document.getElementById("canvas");
	var context = canvas.getContext('2d');
}

Canvas画圆


							context.arc(
  centerx, centery,
  radius,
  startingAngle, endingAngle,
  anticlockwise = false
)

常用CanvasAPI


cxt.clearRect(0, 0, 1024, 768); // 矩形框内清空内容
cxt.fillStyle = ball.color; // 设置填充颜色

cxt.beginPath(); // 拿起画笔开始绘画

cxt.arc(ball.x, ball.y, ball.r, 0, 2 * Math.PI); // 画一个圆的路径
cxt.closePath(); // 提笔
cxt.fill(); // 填充颜色

一些物理知识（加速度）


							var ball = {x: 512, y: 100, r: 20, g: 2, vx: -4, vy: -10, color: "#005588"};

function update() {
	ball.x += ball.vx;
	ball.y += ball.vy;
	ball.vy += ball.g;

	if (ball.y >= 768 - ball.r) {
		ball.y = 768 - ball.r;
		ball.vy = -ball.vy / 2;
	}
}
